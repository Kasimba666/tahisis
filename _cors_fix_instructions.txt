Инструкция — исправление CORS в Nginx api.tarihgis.ru

Замените весь блок server {} в Fastpanel на этот:

server {
    server_name api.tarihgis.ru;

    listen 95.163.242.153:80;
    listen 95.163.242.153:443 ssl;

    ssl_certificate "/var/www/httpd-cert/api.tarihgis.ru_2026-05-30-11-42_07.crt";
    ssl_certificate_key "/var/www/httpd-cert/api.tarihgis.ru_2026-05-30-11-42_07.key";

    # Разрешённые origin'ы для CORS
    map $http_origin $cors_origin {
        default "";
        "https://tarihgis.ru" "$http_origin";
        "http://localhost:5173" "$http_origin";
    }

    location / {
        # Ключевые настройки для решения проблемы 502
        proxy_ssl_server_name on;
        proxy_ssl_name $host;

        proxy_pass https://supabase-proxy.kasimba666.workers.dev;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";

        # === CORS headers ===
        # Убираем дублирующиеся заголовки от бэкенда/воркера
        proxy_hide_header 'Access-Control-Allow-Origin';
        proxy_hide_header 'Access-Control-Allow-Methods';
        proxy_hide_header 'Access-Control-Allow-Headers';
        proxy_hide_header 'Access-Control-Allow-Credentials';
        proxy_hide_header 'Access-Control-Max-Age';

        add_header 'Access-Control-Allow-Origin' $cors_origin always;
        add_header 'Access-Control-Allow-Methods' 'GET, POST, OPTIONS, PUT, PATCH, DELETE' always;
        add_header 'Access-Control-Allow-Headers' 'x-supabase-api-version, x-client-info, accept-profile, x-retry-count, authorization, apikey, content-type' always;
        add_header 'Access-Control-Allow-Credentials' 'true' always;

        # Handle preflight (OPTIONS) requests
        if ($request_method = 'OPTIONS') {
            add_header 'Access-Control-Allow-Origin' $cors_origin always;
            add_header 'Access-Control-Allow-Methods' 'GET, POST, OPTIONS, PUT, PATCH, DELETE' always;
            add_header 'Access-Control-Allow-Headers' 'x-supabase-api-version, x-client-info, accept-profile, x-retry-count, authorization, apikey, content-type' always;
            add_header 'Access-Control-Allow-Credentials' 'true' always;
            add_header 'Access-Control-Max-Age' 86400 always;
            return 204;
        }
        # ================================
        
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
    }

    error_log /var/www/tarihgis_ru_usr/data/logs/api.tarihgis.ru-frontend.error.log;
    access_log /var/www/tarihgis_ru_usr/data/logs/api.tarihgis.ru-frontend.access.log;
}

После замены — нажмите Сохранить и перезапустите Nginx (кнопка в Fastpanel или `systemctl reload nginx` по SSH).
